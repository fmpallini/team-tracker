**Goal:** Implement the v1.1 feedback round: onboarding, visual polish, org-chart tree, editor upgrades, closed risks / item notes / milestone followups (schema v2), calendar fixes, global help, and three bug fixes (spec: `docs/superpowers/specs/2026-07-05-v1.1-feedback-design.md`).

**Tech Stack:** TypeScript strict, esbuild, vitest + jsdom. No new dependencies allowed.

## Global Constraints

- Zero runtime dependencies — dev-only tooling stays `esbuild`, `typescript`, `vitest`, `jsdom`.
- Every user-visible string goes through `t(locale, key)`; every new key added to BOTH `pt-BR` and `en-US` maps in `src/core/i18n.ts`.
- All content mutations via `store.update(fn)`; nav-only via `store.updateNav(fn)`.
- Feature-detect browser APIs absent in jsdom (`CSS.highlights`, fullscreen, FS Access) — code must degrade gracefully and stay testable.
- Each behavior change gets a vitest test in the matching `test/<module>.test.ts`; bugs get a regression test written BEFORE the fix.
- `npm run typecheck` and `npm test` must pass at every commit.

export function teamHasHistory(store: Store, teamId: string): boolean {
  return store.doc.nav.panes.some((p) => p.history.some((loc) => loc.teamId === teamId))
}

export function openTeamDefaultLayout(pm: PaneManager, store: Store, teamId: string): void {
  store.updateNav((d) => { d.nav.split = true; d.nav.focusedPane = 0 })
  pm.openInPane(1, { teamId, ref: { kind: 'members' } })
  pm.openInPane(0, { teamId, ref: { kind: 'daily', date: todayIso() } })
}

function selectTeam(id: string): void {
